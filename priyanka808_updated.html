import pandas as pd

housing_data = pd.read_csv('Housing.csv')

average_price = housing_data['price'].mean()

max_price = housing_data['price'].max()
min_price = housing_data['price'].min()


house_max_area = housing_data.loc[housing_data['area'].idxmax()]

average_bedrooms = housing_data['bedrooms'].mean()
average_bathrooms = housing_data['bathrooms'].mean()

common_bedrooms = housing_data['bedrooms'].mode()[0]
common_bathrooms = housing_data['bathrooms'].mode()[0]

num_houses_with_ac = housing_data[housing_data['airconditioning'] == 'yes'].shape[0]

num_houses_with_guestroom = housing_data[housing_data['guestroom'] == 'yes'].shape[0]

num_houses_with_basement = housing_data[housing_data['basement'] == 'yes'].shape[0]

average_price_furnished = housing_data[housing_data['furnishingstatus'] == 'furnished']['price'].mean()
average_price_semi_furnished = housing_data[housing_data['furnishingstatus'] == 'semi-furnished']['price'].mean()
average_price_unfurnished = housing_data[housing_data['furnishingstatus'] == 'unfurnished']['price'].mean()

print("Important Data from the Housing Project:")

print(f"2. Maximum House Price: {max_price}")
print(f"3. Minimum House Price: {min_price}")
print(f"4. House with Maximum Area:\n{house_max_area}")
print(f"5. Average Number of Bedrooms: {average_bedrooms}")
print(f"6. Average Number of Bathrooms: {average_bathrooms}")
print(f"7. Most Common Number of Bedrooms: {common_bedrooms}")
print(f"8. Most Common Number of Bathrooms: {common_bathrooms}")
print(f"9. Number of Houses with Air Conditioning: {num_houses_with_ac}")
print(f"10. Number of Houses with Guestroom: {num_houses_with_guestroom}")
print(f"11. Number of Houses with Basement: {num_houses_with_basement}")
